02 Data Scramble ascii noise · decryption 

/* JS randomises each char through ASCII before resolving.
   Trigger on mount + loop every 5s. */
const CHARS = '!@#$%^&*01アイウエオカキ';

function scramble(el, finalText) {
  let iter = 0;
  const interval = setInterval(() => {
    el.querySelectorAll('.ch').forEach((ch, i) => {
      if (i < iter) {
        ch.textContent = finalText[i];
        ch.classList.remove('scrambling');
      } else {
        ch.textContent = CHARS[Math.floor(Math.random()*CHARS.length)];
        ch.classList.add('scrambling');
      }
    });
    if (++iter > finalText.length) clearInterval(interval);
  }, 60);
}

/* CSS for scrambling state */
.ch.scrambling {
  color: rgba(0,204,255,.9);
  text-shadow: 0 0 8px rgba(0,204,255,.9);
}
